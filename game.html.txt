    <script>
        const questions = [
            { text: "I have lived here for ten years.", type: "present" },
            { text: "She visited Paris in 2019.", type: "past" },
            { text: "We haven't finished the report yet.", type: "present" },
            { text: "They went to the cinema last night.", type: "past" },
            { text: "Have you seen a ghost yet?", type: "present" },
            { text: "My grandfather was a doctor.", type: "past" },
            { text: "I just ate a huge sandwich.", type: "past" },
            { text: "Science has come a long way.", type: "present" },
            { text: "He has started a new job.", type: "present" },
            { text: "We have already seen this movie.", type: "present" },
            { text: "She lost her keys yesterday morning.", type: "past" },
            { text: "We lived in Tokyo when I was a child.", type: "past" },
            { text: "Have they eaten their dinner yet?", type: "present" },
        ];

        let currentIdx = 0;
        let score = 0;
        const card = document.getElementById('card');
        const scoreEl = document.getElementById('score');

        // Khởi tạo thẻ đầu tiên
        function loadCard() {
            if (currentIdx >= questions.length) {
                card.innerHTML = "🎉 Well done!<br>Điểm của bạn: " + score;
                card.style.cursor = "default";
                return;
            }
            card.textContent = questions[currentIdx].text;
            card.style.transform = "translate(0, -50%)";
            card.style.backgroundColor = "white";
        }

        let isDragging = false;
        let startX = 0;

        card.addEventListener('mousedown', startDrag);
        card.addEventListener('touchstart', startDrag);

        function startDrag(e) {
            if (currentIdx >= questions.length) return;
            isDragging = true;
            startX = e.type === 'mousedown' ? e.clientX : e.touches[0].clientX;
            card.style.transition = "none";
        }

        window.addEventListener('mousemove', drag);
        window.addEventListener('touchmove', drag);

        function drag(e) {
            if (!isDragging) return;
            const x = e.type === 'mousemove' ? e.clientX : e.touches[0].clientX;
            const moveX = x - startX;
            card.style.transform = `translate(${moveX}px, -50%) rotate(${moveX * 0.1}deg)`;
            
            if (moveX < -50) card.style.backgroundColor = "#fadbd8";
            else if (moveX > 50) card.style.backgroundColor = "#d4efdf";
            else card.style.backgroundColor = "white";
        }

        window.addEventListener('mouseup', endDrag);
        window.addEventListener('touchend', endDrag);

        function endDrag(e) {
            if (!isDragging) return;
            isDragging = false;
            const moveX = parseInt(card.style.transform.match(/translate\((.*?)px/)[1]);
            
            if (moveX < -150) { // Chọn Past Simple
                checkAnswer("past");
            } else if (moveX > 150) { // Chọn Present Perfect
                checkAnswer("present");
            } else {
                card.style.transition = "transform 0.3s ease";
                card.style.transform = "translate(0, -50%)";
            }
        }

        function checkAnswer(choice) {
            const correct = questions[currentIdx].type === choice;
            if (correct) score++;
            
            scoreEl.textContent = `Điểm: ${score}`;
            currentIdx++;
            
            card.style.transition = "transform 0.5s ease";
            card.style.transform = choice === "past" ? "translate(-600px, -50%)" : "translate(600px, -50%)";
            
            setTimeout(loadCard, 500);
        }

        loadCard();
    </script>
